polish_modifier = {
	every_scope_state = {
		limit = {
			state_region = { is_homeland = cu:polish }
		}
		add_modifier = {
			name = $MODIFIER$
		}
	}
}

commonwealth_modifier = {
	every_scope_state = {
		limit = {
			or = {
				state_region = { is_homeland = cu:polish }
				state_region = { is_homeland = cu:lithuanian }
				state_region = { is_homeland = cu:byelorussian }
				state_region = s:STATE_ZHYTOMYR
				state_region = s:STATE_PODOLIA
			}
		}
		add_modifier = {
			name = $MODIFIER$
		}
	}
}
